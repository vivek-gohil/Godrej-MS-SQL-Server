CREATE TABLE Employee
(
    EmpId INT PRIMARY KEY,
    EmpName VARCHAR(50),
    Department VARCHAR(30),
    City VARCHAR(30),
    Salary DECIMAL(10,2),
    JoiningDate DATE
);
